
startup_variables_unique_mechanics_effect = {
	if = {
		limit = {
			tag = ATH
			has_dlc = "Magna Graecia"
		}
		set_variable = {
			name = ath_unique_mech
		}
	}
	if = {
		limit = {
			tag = SPA
			has_dlc = "Magna Graecia"
		}
		set_variable = {
			name = spa_unique_mech
		}	
	}
	if = {
		limit = {
			tag = SYR
			has_dlc = "Magna Graecia"
		}
		set_variable = {
			name = syr_unique_mech	
		}
	}
	if = {
		limit = {
			tag = EPI
			has_dlc = "Epirus Flavor Pack"
		}
		set_variable = {
			name = epi_unique_mech
		}
	}
}
