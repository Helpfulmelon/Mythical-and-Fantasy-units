is_diadochus_trigger = {
	OR = {
		tag = MAC
		tag = EGY
		tag = PRY
		tag = SEL
		tag = TRE
	}
}

pyrrhus_wife_finder_trigger = {
	any_living_character = {
		is_ruler = no
		is_female = yes
		age >= 16
		age <= 30
		is_married = no
		prisoner = no
		from_ruler_family = yes
		employer = root
		is_at_location = root.capital_scope
	}
}

		AND = {
			tag = MAC
